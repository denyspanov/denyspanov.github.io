Search docker images:
{% highlight shell %}
docker search <name>
{% endhighlight %}
Run container in background:
{% highlight shell %}
docker run -d <name>
{% endhighlight %}
List all running containers:
{% highlight shell %}
docker ps
{% endhighlight %}
More details about running container:
{% highlight shell %}
docker inspect <name/container-id>
{% endhighlight %}
Show container logs:
{% highlight shell %}
docker logs <name/container-id>
{% endhighlight %}
Expose ports:
{% highlight shell %}
docker run -d --name <container name> -p <port>:<port> <container-image>
docker run -d --name redisHostPort -p 6379:6379 redis:latest
docker run -d --name redisHostPort -p 127.0.0.1:6379:6379 redis:latest
{% endhighlight %}
